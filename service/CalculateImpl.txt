package cashflowcalculations.serviceimpl;import java.util.Map;import cashflowcalculations.model.CashFlow;import cashflowcalculations.service.Calculate;import cashflowcalculations.service.ExpenditureAccumulation;import cashflowcalculations.service.IncomeAccumulation;public class CalculateImpl implements Calculate {	@Override	public CashFlow calculateCashFlow() {		CashFlow cashFlow = new CashFlow();		IncomeAccumulation incomeAccumulation = new IncomeAccumulationImpl();		ExpenditureAccumulation expenditureAccumulation = new ExpenditureAccumulationImpl();		double finalIncomeValue = incomeAccumulation.findTotalIncome();		cashFlow = expenditureAccumulation.findTotalExpenditure();		double cashFlowValue = finalIncomeValue - cashFlow.getExpenditureDetails().getTotalExpenditure();		cashFlow.setAmount(cashFlowValue);		return cashFlow;	}}
